### الفصل 2: إنشاء نموذج الملاحظة

#### ما هو Dataclass؟

Dataclass في Python هو طريقة مختصرة لإنشاء فئات تهدف إلى تخزين البيانات. بدلاً من كتابة الكثير من الكود المتكرر، يمكننا استخدام `@dataclass` decorator.

#### التنفيذ العملي

أنشئ ملف `src/models.py`:

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Optional

@dataclass
class Note:
    """نموذج يمثل ملاحظة واحدة.
    
    هذا النموذج يحتوي على جميع المعلومات الخاصة بالملاحظة:
    - العنوان: اسم الملاحظة
    - المحتوى: نص الملاحظة بصيغة Markdown
    - مسار الملف: الموقع حيث يتم حفظ الملاحظة
    - تاريخ الإنشاء: متى تم إنشاء الملاحظة
    - تاريخ التحديث: متى تم آخر تعديل للملاحظة
    """
    
    # الحقول المطلوبة
    title: str                    # عنوان الملاحظة
    content: str                  # محتوى الملاحظة
    file_path: str                # مسار الملف
    created_at: datetime          # تاريخ الإنشاء
    updated_at: datetime          # تاريخ التحديث
    
    # حقول اختيارية
    tags: Optional[list[str]] = None   # الوسوم المرتبطة بالملاحظة
    
    def __str__(self) -> str:
        """تمثيل نصي للملاحظة - مفيد للتصحيح."""
        return f"Note(title='{self.title}', path='{self.file_path}')"
    
    def to_dict(self) -> dict:
        """تحويل الملاحظة إلى قاموس."""
        return {
            'title': self.title,
            'content': self.content,
            'file_path': self.file_path,
            'created_at': self.created_at.isoformat(),
            'updated_at': self.updated_at.isoformat(),
            'tags': self.tags or []
        }
    
    @property
    def word_count(self) -> int:
        """حساب عدد الكلمات في الملاحظة."""
        return len(self.content.split())
    
    @property
    def char_count(self) -> int:
        """حساب عدد الأحرف في الملاحظة."""
        return len(self.content)
```

#### شرح الكود:

1. **@dataclass**: هذا decorator يولد تلقائياً:
   - `__init__` method
   - `__repr__` method
   - `__eq__` method

2. **حقول البيانات**:
   - `title`: عنوان الملاحظة
   - `content`: نص الملاحظة
   - `file_path`: مسار الملف
   - `created_at`, `updated_at`: تواريخ التعقب

3. **الخصائص (Properties)**:
   - `word_count`: عدد الكلمات
   - `char_count`: عدد الأحرف

### الفصل 3: إدارة الملفات

#### إنشاء ملف إدارة الملفات

أنشئ `src/file_manager.py`:

```python
import os
from datetime import datetime
from pathlib import Path
from typing import Optional
from src.models import Note

def save_note(note: Note, root_dir: str = "./data") -> bool:
    """حفظ ملاحظة إلى ملف Markdown.
    
    Args:
        note: كائن الملاحظة المراد حفظها
        root_dir: المجلد الرئيسي لحفظ الملفات
        
    Returns:
        True إذا تم الحفظ بنجاح، False في حالة الفشل
    """
    try:
        # إنشاء المسار الكامل
        full_path = Path(root_dir) / note.file_path
        
        # إنشاء المجلدات إذا لم تكن موجودة
        full_path.parent.mkdir(parents=True, exist_ok=True)
        
        # حفظ الملف بترميز UTF-8 لدعم العربية
        with open(full_path, 'w', encoding='utf-8') as f:
            f.write(note.content)
        
        print(f"✅ تم حفظ الملاحظة: {note.title}")
        return True
        
    except Exception as e:
        print(f"❌ خطأ في حفظ الملاحظة: {e}")
        return False

def read_note(file_path: str, root_dir: str = "./data") -> Optional[Note]:
    """قراءة ملاحظة من ملف Markdown.
    
    Args:
        file_path: مسار الملف
        root_dir: المجلد الرئيسي
        
    Returns:
        كائن Note أو None إذا لم يتم العثور على الملف
    """
    try:
        full_path = Path(root_dir) / file_path
        
        # التحقق من وجود الملف
        if not full_path.exists():
            print(f"⚠️  الملف غير موجود: {file_path}")
            return None
        
        # قراءة الملف بترميز UTF-8
        with open(full_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # الحصول على معلومات الملف
        stat = full_path.stat()
        
        # إنشاء كائن الملاحظة
        note = Note(
            title=Path(file_path).stem,
            content=content,
            file_path=file_path,
            created_at=datetime.fromtimestamp(stat.st_ctime),
            updated_at=datetime.fromtimestamp(stat.st_mtime)
        )
        
        return note
        
    except Exception as e:
        print(f"❌ خطأ في قراءة الملاحظة: {e}")
        return None

def update_note(note: Note, root_dir: str = "./data") -> bool:
    """تحديث ملاحظة موجودة.
    
    Args:
        note: كائن الملاحظة المحدث
        root_dir: المجلد الرئيسي
        
    Returns:
        True إذا تم التحديث بنجاح
    """
    # تحديث تاريخ التعديل
    note.updated_at = datetime.now()
    
    # إعادة حفظ الملاحظة
    return save_note(note, root_dir)

def delete_note(file_path: str, root_dir: str = "./data") -> bool:
    """حذف ملاحظة.
    
    Args:
        file_path: مسار الملف
        root_dir: المجلد الرئيسي
        
    Returns:
        True إذا تم الحذف بنجاح
    """
    try:
        full_path = Path(root_dir) / file_path
        
        # التحقق من وجود الملف
        if not full_path.exists():
            print(f"⚠️  الملف غير موجود: {file_path}")
            return False
        
        # التحقق من أنه ملف (ليس مجلد)
        if not full_path.is_file():
            print(f"❌ المسار ليس ملفاً: {file_path}")
            return False
        
        # حذف الملف
        full_path.unlink()
        print(f"✅ تم حذف الملاحظة: {file_path}")
        return True
        
    except Exception as e:
        print(f"❌ خطأ في حذف الملاحظة: {e}")
        return False
```

#### شرح الكود:

1. **save_note()**:
   - ينشئ المسار الكامل للملف
   - يُنشئ المجلدات إذا لزم الأمر
   - يحفظ المحتوى بترميز UTF-8

2. **read_note()**:
   - يتحقق من وجود الملف
   - يقرأ المحتوى بترميز UTF-8
   - يستخرج معلومات الملف (التواريخ)

3. **update_note()**:
   - يُحدث `updated_at`
   - يعيد حفظ الملاحظة

4. **delete_note()**:
   - يتحقق من وجود الملف
   - يحذف الملف بأمان

### الفصل 4: مسح المجلدات

#### إنشاء ملف المسح الضوئي

أنشئ `src/scanner.py`:

```python
import os
from pathlib import Path
from typing import List
from src.models import Note
from src.file_manager import read_note

def get_all_notes(root_dir: str = "./data") -> List[Note]:
    """الحصول على جميع الملاحظات في المجلد.
    
    هذه الدالة تمسح جميع المجلدات الفرعية وتجمع كل ملفات Markdown.
    
    Args:
        root_dir: المجلد الرئيسي للبحث
        
    Returns:
        قائمة بجميع الملاحظات الموجودة
    """
    notes = []
    root_path = Path(root_dir)
    
    # التحقق من وجود المجلد
    if not root_path.exists():
        print(f"⚠️  المجلد غير موجود: {root_dir}")
        return notes
    
    # المسح المتكرر لجميع الملفات
    for md_file in root_path.rglob("*.md"):
        # تخطي الملفات المخفية
        if md_file.name.startswith('.'):
            continue
        
        try:
            # الحصول على المسار النسبي
            rel_path = md_file.relative_to(root_dir)
            
            # قراءة الملاحظة
            note = read_note(str(rel_path), root_dir)
            if note:
                notes.append(note)
                print(f"✅ تم العثور: {note.title}")
                
        except Exception as e:
            print(f"⚠️  خطأ في قراءة {md_file}: {e}")
            continue
    
    print(f"\n📊 الإجمالي: {len(notes)} ملاحظة")
    return notes

def get_notes_by_folder(root_dir: str = "./data") -> dict:
    """تنظيم الملاحظات حسب المجلد.
    
    Args:
        root_dir: المجلد الرئيسي
        
    Returns:
        قاموس: {folder_name: [notes]}
    """
    notes_dict = {}
    notes = get_all_notes(root_dir)
    
    for note in notes:
        # استخراج المجلد
        folder = str(Path(note.file_path).parent)
        if folder == ".":
            folder = "الرئيسي"
        
        if folder not in notes_dict:
            notes_dict[folder] = []
        
        notes_dict[folder].append(note)
    
    return notes_dict
```

### الفصل 5: تحليل Markdown

#### ما هو Markdown؟

Markdown هو لغة ترميز خفيفة تُستخدم لكتابة النصوص المنسقة بطريقة سهلة.

#### إنشاء محلل Markdown

أنشئ `src/parser.py`:

import markdown
from typing import Optional

def render_html(markdown_text: str) -> str:
    """تحويل نص Markdown إلى HTML.
    
    Args:
        markdown_text: نص Markdown المراد تحويله
        
    Returns:
        نص HTML الناتج
    """
    try:
        # إنشاء محول Markdown مع الإضافات
        md = markdown.Markdown(
            extensions=[
                'extra',           # ميزات إضافية
                'codehilite',      # تمييز الكود
                'toc',             # جدول المحتويات
                'nl2br',           # تحويل الأسطر الجديدة
            ]
        )
        
        # تحويل النص
        html = md.convert(markdown_text)
        return html
        
    except Exception as e:
        print(f"❌ خطأ في تحليل Markdown: {e}")
        return f"<p>{markdown_text}</p>"

    </style>
    """
    
    # دمج CSS
    final_css = css_style + default_css if css_style else default_css
    
    # إنشاء صفحة HTML كاملة
    full_html = f"""
    <!DOCTYPE html>
    <html dir="rtl" lang="ar">
    <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>معاينة</title>
        {final_css}
    </head>
    <body>
        {html_content}
    </body>
    </html>
    """
    
    return full_html

from typing import List
from src.models import Note

def search_notes(query: str, notes: List[Note]) -> List[Note]:
    """البحث في الملاحظات.
    
    Args:
        query: كلمة البحث
        notes: قائمة الملاحظات
        
    Returns:
        الملاحظات المطابقة
    """
    if not query:
        return notes
    
    query_lower = query.lower()
    results = []
    
    for note in notes:
        # البحث في العنوان
        if query_lower in note.title.lower():
            results.append(note)
            continue
        
        # البحث في المحتوى
        if query_lower in note.content.lower():
            results.append(note)
    
    return results

def search_notes_advanced(query: str, notes: List[Note], 
                         case_sensitive: bool = False) -> List[Note]:
    """بحث متقدم مع خيارات إضافية.
    
    Args:
        query: كلمة البحث
        notes: قائمة الملاحظات
        case_sensitive: البحث الحساس لحالة الأحرف
        
    Returns:
        الملاحظات المطابقة
    """
    if not query:
        return notes
    
    search_query = query if case_sensitive else query.lower()
    results = []
    
    for note in notes:
        title = note.title if case_sensitive else note.title.lower()
        content = note.content if case_sensitive else note.content.lower()
        
        if search_query in title or search_query in content:
            results.append(note)
    
    return results

#### إنشاء ملف الوسوم

أنشئ `src/tags.py`:

```python
import re
from typing import List, Set
from src.models import Note

def extract_tags(text: str) -> List[str]:
    """استخراج الوسوم من النص.
    
    الوسوم هي الكلمات التي تبدأ بـ #
    مثال: #python #برمجة #تعلم
    
    Args:
        text: النص المراد استخراج الوسوم منه
        
    Returns:
        قائمة بالوسوم الفريدة مرتبة أبجدياً
    """
    # البحث عن جميع الأنماط #tag
    pattern = r'#([\w\u0600-\u06FF-]+)'
    tags = re.findall(pattern, text)
    
    # إزالة التكرار وترتيبها
    unique_tags = sorted(set(tags))
    return unique_tags

def get_all_tags(notes: List[Note]) -> List[str]:
    """الحصول على جميع الوسوم من جميع الملاحظات.
    
    Args:
        notes: قائمة الملاحظات
        
    Returns:
        قائمة بجميع الوسوم الفريدة
    """
    all_tags: Set[str] = set()
    
    for note in notes:
        tags = extract_tags(note.content)
        all_tags.update(tags)
    
    return sorted(all_tags)

def filter_by_tag(tag: str, notes: List[Note]) -> List[Note]:
    """تصفية الملاحظات حسب وسم معين.
    
    Args:
        tag: الوسم المطلوب
        notes: قائمة الملاحظات
        
    Returns:
        الملاحظات التي تحتوي على الوسم
    """
    filtered = []
    
    for note in notes:
        if tag in extract_tags(note.content):
            filtered.append(note)
    
    return filtered

def filter_by_tags(tags: List[str], notes: List[Note], 
                   mode: str = 'AND') -> List[Note]:
    """تصفية الملاحظات حسب عدة وسوم.
    
    Args:
        tags: قائمة الوسوم
        notes: قائمة الملاحظات
        mode: 'AND' (جميع الوسوم) أو 'OR' (أي وسم)
        
    Returns:
        الملاحظات المطابقة للشرط
    """
    if mode == 'AND':
        # الملاحظة يجب أن تحتوي على جميع الوسوم
        filtered = []
        for note in notes:
            note_tags = extract_tags(note.content)
            if all(tag in note_tags for tag in tags):
                filtered.append(note)
        return filtered
    
    else:  # OR mode
        # الملاحظة تحتوي على أي وسم
        filtered = []
        for note in notes:
            note_tags = extract_tags(note.content)
            if any(tag in note_tags for tag in tags):
                filtered.append(note)
        return filtered
```

```python
import customtkinter as ctk
from src.scanner import get_all_notes

class LocalNoteApp:
    """التطبيق الرئيسي."""
    
    def __init__(self):
        """تهيئة التطبيق."""
        # إعداد المظهر
        ctk.set_appearance_mode("dark")
        ctk.set_default_color_theme("blue")
        
        # إنشاء النافذة
        self.root = ctk.CTk()
        self.root.title("الملاحظة المحلية - Local Note")
        self.root.minsize(900, 600)
        self.root.geometry("1100x700")
        
        # إعداد التخطيط
        self.setup_layout()
        
        # تحميل الملاحظات
        self.notes = get_all_notes()
        
        # إنشاء المكونات
        self.create_widgets()
        
        # التعامل مع إغلاق النافذة
        self.root.protocol("WM_DELETE_WINDOW", self.on_closing)
    
    def setup_layout(self):
        """إعداد تخطيط النافذة."""
        # عمودين: الشريط الجانبي (30%) والمحتوى الرئيسي (70%)
        self.root.grid_columnconfigure(0, weight=1)
        self.root.grid_columnconfigure(1, weight=3)
        self.root.grid_rowconfigure(0, weight=1)
    
    def create_widgets(self):
        """إنشاء مكونات الواجهة."""
        # الشريط الجانبي
        self.sidebar = ctk.CTkFrame(self.root)
        self.sidebar.grid(row=0, column=0, sticky="nsew", padx=5, pady=5)
        
        # منطقة المحتوى الرئيسية
        self.main_content = ctk.CTkFrame(self.root)
        self.main_content.grid(row=0, column=1, sticky="nsew", padx=5, pady=5)
        
        # إنشاء محتوى الشريط الجانبي
        self.create_sidebar_content()
        
        # إنشاء محتوى المنطقة الرئيسية
        self.create_main_content()
    
    def create_sidebar_content(self):
        """إنشاء محتوى الشريط الجانبي."""
        # عنوان
        title = ctk.CTkLabel(
            self.sidebar,
            text="📝 الملاحظات",
            font=("", 20, "bold")
        )
        title.pack(pady=10)
        
        # مربع البحث
        self.search_var = ctk.StringVar()
        search_entry = ctk.CTkEntry(
            self.sidebar,
            placeholder_text="🔍 بحث...",
            textvariable=self.search_var
        )
        search_entry.pack(fill="x", padx=10, pady=5)
        
        # قائمة الملاحظات
        self.notes_listbox = ctk.CTkTextbox(
            self.sidebar,
            height=400
        )
        self.notes_listbox.pack(fill="both", expand=True, padx=10, pady=5)
        
        # تحديث القائمة
        self.update_notes_list()
    
    def create_main_content(self):
        """إنشاء المحتوى الرئيسي."""
        # شريط الأدوات
        toolbar = ctk.CTkFrame(self.main_content)
        toolbar.pack(fill="x", padx=5, pady=5)
        
        # زر جديد
        new_btn = ctk.CTkButton(
            toolbar,
            text="➕ ملاحظة جديدة",
            command=self.new_note
        )
        new_btn.pack(side="left", padx=5)
        
        # زر الحفظ
        save_btn = ctk.CTkButton(
            toolbar,
            text="💾 حفظ",
            command=self.save_note
        )
        save_btn.pack(side="left", padx=5)
        
        # زر الحذف
        delete_btn = ctk.CTkButton(
            toolbar,
            text="🗑️ حذف",
            command=self.delete_note
        )
        delete_btn.pack(side="left", padx=5)
        
        # محرر النصوص
        self.editor = ctk.CTkTextbox(
            self.main_content,
            font=("Consolas", 14),
            wrap="word"
        )
        self.editor.pack(fill="both", expand=True, padx=5, pady=5)
    
    def update_notes_list(self):
        """تحديث قائمة الملاحظات."""
        self.notes_listbox.delete("1.0", "end")
        
        for note in self.notes:
            self.notes_listbox.insert("end", f"📄 {note.title}\n")
    
    def new_note(self):
        """إنشاء ملاحظة جديدة."""
        print("إنشاء ملاحظة جديدة...")
    
    def save_note(self):
        """حفظ الملاحظة الحالية."""
        print("حفظ الملاحظة...")
    
    def delete_note(self):
        """حذف الملاحظة الحالية."""
        print("حذف الملاحظة...")
    
    def on_closing(self):
        """معالجة إغلاق النافذة."""
        self.root.destroy()
    
    def run(self):
        """تشغيل التطبيق."""
        self.root.mainloop()

# نقطة الدخول
if __name__ == "__main__":
    app = LocalNoteApp()
    app.run()
```
